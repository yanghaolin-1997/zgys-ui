• 快速开始
    • OperateBtn 操作按钮

    # 快速开始

    # 安装组件库

    npm i zgys-ui

    # 引用组件库

    # 全部引入

    import 'zgys-ui/lib/css/index.css'
import ZgysUI from 'zgys-ui'
Vue.use(ZgysUI)

    # 按需引用

    # 1.安装插件

    npm i babel-plugin-import -D

    # 2.配置插件

    // babel.config.js
{
  plugins: [
    [
      'import',
      {
        libraryName: 'zgys-ui',
        style: name => {
          const n = name.split('/')[2].split('.')[0]
          return `zgys-ui/lib/css/other/${n}.css`
        },
        customName: name => {
          return `zgys-ui/lib/${name}.umd.js`
        }
      }
    ]
  ]
}

    # 3.引入组件

    import { OperateBtn } from 'zgys-ui'
Vue.use(OperateBtn)
